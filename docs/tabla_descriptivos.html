descriptivos <- 
  transectos %>% inner_join(densidad_avg) %>%
  dplyr::select(-sd) %>%
  mutate(mean = round(mean, 2),
         se = round(se,2)) %>%
  unite("density", mean,se, sep = " ± ") %>% 
  inner_join(diversidad_avg) %>%
  dplyr::select(-sd) %>%
  mutate(mean = round(mean, 2),
         se = round(se,2)) %>%
  unite("diversity", mean,se, sep = " ± ") %>%
  inner_join(richness) %>%
  rename(richness = riq) %>%
  inner_join(abundancia) %>%
  dplyr::select(-sd) %>%
