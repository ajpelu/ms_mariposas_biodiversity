den <- raster::raster(here::here("data/density.ovr"))
riq <- raster::raster(here::here("data/richness.ovr"))
div <- raster::raster(here::here("data/diversity_v2.ovr"))

s <- stack(den, riq, div)

df <- as.data.frame(s, xy=TRUE) %>% 
  filter(!is.na(density))
